✅ STEP 2 — Create NFC API endpoint (CORE)

Create a file:

nfc_api.py
Example:
from rest_framework.decorators import api_view
from rest_framework.response import Response
from stock.models import StockItem

@api_view(['POST'])
def nfc_scan(request):
    uid = request.data.get("uid", "").strip().upper()

    item = StockItem.objects.filter(metadata__nfc_tag=uid).first()

    if not item:
        return Response({
            "status": "not_found",
            "uid": uid,
            "action": "register"
        })

    return Response({
        "status": "found",
        "item_id": item.pk,
        "actions": ["view", "pick", "restock"]
    })
